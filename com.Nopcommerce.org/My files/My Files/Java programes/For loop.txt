package javapractice;

public class ForLoopExample {

	public static void main(String[] args) {

		System.out.println("*********************");
		System.out.println();

		for (int i = 1; i <= 10; i++) {

			System.out.println(i + " IT Industry");

		}

		System.out.println();
		System.out.println("*********************");

	}

}
