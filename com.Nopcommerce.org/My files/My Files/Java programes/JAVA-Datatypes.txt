public class DataTypesExample7{
		
	boolean b;
	char c;
	byte by;
	short s;
	int i;
	long l;
	float f;
	double d;
		
	public static void main(String[] args){
		
		DataTypesExample7 dte = new DataTypesExample7();
		System.out.println(dte.b);
		System.out.println(dte.c);
		System.out.println(dte.by);
		System.out.println(dte.s);
		System.out.println(dte.i);
		System.out.println(dte.l);
		System.out.println(dte.f);
		System.out.println(dte.d);
	
	}

}
